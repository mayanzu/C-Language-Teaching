printf函数详解

printf基本用法

printf 是 C 语言中最常用的输出函数，用于向屏幕输出格式化的文本。它是 "print formatted" 的缩写，定义在 stdio.h 头文件中。

基本语法

printf("格式字符串", 参数列表);

  • 格式字符串：包含要输出的文本和格式说明符（如 %d、%f 等）
  • 参数列表：要输出的变量，按照格式说明符的顺序提供
printf基本示例
// printf基本用法示例
#include <stdio.h>

int main() {
    // 简单文本输出 - 不需要格式说明符
    printf("Hello, World!\n");

    // 格式化输出整数
    int age = 20;
    printf("年龄: %d\n", age);  // %d是整数的格式说明符

    // 格式化输出浮点数
    float height = 1.75;
    printf("身高: %.2f 米\n", height);  // %.2f保留2位小数

    // 格式化输出字符
    char grade = 'A';
    printf("等级: %c\n", grade);  // %c是字符的格式说明符

    // 一次输出多个变量
    int score = 95;
    printf("学生年龄%d岁，成绩%d分，等级%c\n", age, score, grade);

    return 0;
}
重要提示：

printf 的格式说明符数量必须与后面的参数数量一致，类型也必须匹配。\n 表示换行，如果不加 \n，下一次输出会接在同一行。

常用格式说明符

格式说明符决定了数据以什么形式输出，是 printf 最核心的知识点。

格式符 含义 示例 输出结果
%d 十进制整数 printf("%d", 42); 42
%f 浮点数（默认6位小数） printf("%f", 3.14); 3.140000
%.2f 保留2位小数 printf("%.2f", 3.14159); 3.14
%c 单个字符 printf("%c", 'A'); A
%s 字符串 printf("%s", "hello"); hello
%% 输出百分号本身 printf("100%%"); 100%
格式说明符实战示例
// 各种格式说明符的使用示例
#include <stdio.h>

int main() {
    int age = 18;
    float score = 95.5;
    char grade = 'A';
    char name[20] = "小明";

    // 基本格式输出
    printf("姓名: %s\n", name);
    printf("年龄: %d 岁\n", age);
    printf("成绩: %.1f 分\n", score);
    printf("等级: %c 级\n", grade);

    // 多个变量一起输出
    printf("%s今年%d岁，成绩%.1f分，等级%c\n", name, age, score, grade);

    // 输出百分号
    printf("完成度: 100%%\n");

    return 0;
}

宽度对齐与精度控制

printf 支持多种格式控制，让输出更整齐美观。

宽度与对齐

通过数字指定输出宽度，- 表示左对齐，0 表示前导零填充。

宽度和对齐控制
// 宽度和对齐控制示例
#include <stdio.h>

int main() {
    int num1 = 42, num2 = 123;

    // 宽度控制 - 右对齐（默认）
    printf("右对齐: '%5d'\n", num1);  // '   42'
    printf("右对齐: '%5d'\n", num2);  // '  123'

    // 左对齐
    printf("左对齐: '%-5d'\n", num1);  // '42   '
    printf("左对齐: '%-5d'\n", num2);  // '123  '

    // 前导零
    printf("前导零: '%05d'\n", num1);  // '00042'

    return 0;
}
浮点数精度控制
// 浮点数精度控制示例
#include <stdio.h>

int main() {
    double pi = 3.141592653589793;
    double money = 123.456789;

    // 默认精度（6位小数）
    printf("默认: %f\n", pi);  // 3.141593

    // 指定小数位数
    printf("2位小数: %.2f\n", pi);  // 3.14
    printf("4位小数: %.4f\n", pi);  // 3.1416

    // 货币格式
    printf("金额: %.2f 元\n", money); // 123.46

    return 0;
}

scanf函数详解

scanf基本用法

scanf 是 C 语言中最常用的输入函数，用于从键盘读取用户输入。它是 "scan formatted" 的缩写，定义在 stdio.h 头文件中。

基本语法

scanf("格式字符串", &变量1, &变量2, ...);

  • 格式字符串：指定输入数据的类型和格式
  • &变量：取地址运算符 & 后跟变量名，表示变量的内存地址
scanf基本示例
// scanf基本用法示例
#include <stdio.h>

int main() {
    int age;
    float height;
    char grade;
    char name[50];

    printf("请输入年龄: ");
    scanf("%d", &age);

    printf("请输入身高: ");
    scanf("%f", &height);

    printf("请输入等级: ");
    scanf(" %c", &grade);  // 注意%c前面的空格

    printf("请输入姓名: ");
    scanf("%s", name);  // 字符串不需要&

    printf("\n输入信息:\n");
    printf("年龄: %d\n", age);
    printf("身高: %.2f\n", height);
    printf("等级: %c\n", grade);
    printf("姓名: %s\n", name);

    return 0;
}

地址运算符&

& 是取地址运算符，scanf 需要知道变量在内存中的地址才能把数据存进去。

为什么需要 & ？

scanf 要"存"数据，需要知道"存"在哪里，所以要用 & 获取地址。printf 只是"读"数据，不需要地址。

重要提醒：
  • scanf 读取整数、浮点数、字符时都需要 &
  • 读取字符串时不需要 &（因为数组名本身就是地址）
  • 忘记 & 是最常见的错误，会导致程序崩溃
正确与错误的对比
// 正确与错误的对比
#include <stdio.h>

int main() {
    int num;

    // 错误：缺少&
    // scanf("%d", num);  // 编译错误或程序崩溃！

    // 正确：使用&
    scanf("%d", &num);
    printf("输入的数字是: %d\n", num);

    // 字符串数组名本身就是地址，不需要&
    char str[100];
    scanf("%s", str);  // 正确

    return 0;
}
记忆口诀：

scanf 读数据要地址，& 符号不能少。字符串数组不用 &，数组名就是地址。

scanf常用格式符

scanf 的格式符与 printf 类似，但也有一些不同之处。

格式符 用途 示例
%d 读取整数 scanf("%d", &num);
%f 读取 float 浮点数 scanf("%f", &height);
%lf 读取 double 浮点数 scanf("%lf", &money);
%c 读取单个字符 scanf(" %c", &grade);
%s 读取字符串（遇空格停止） scanf("%s", name);

输入缓冲区问题

scanf 读取数据时，输入的内容会先存入"缓冲区"，scanf 从缓冲区中读取。如果缓冲区中有残留数据，可能导致读取异常。

常见问题：读取整数后再读字符

scanf 读取整数后，输入缓冲区会残留换行符 \n，导致下一个 %c 直接读到换行符。

解决方法：在 %c 前加空格：scanf(" %c", &ch);

常见问题：scanf 不能读取带空格的字符串

scanf 遇到空格就会停止读取。如果要读取一整行，通常使用 fgets()。gets() 已被废弃，不建议使用。

输入缓冲区问题示例
// 输入缓冲区问题示例
#include <stdio.h>

int main() {
    int num;
    char ch;

    printf("请输入一个整数: ");
    scanf("%d", &num);

    // 错误：缓冲区中残留\n，ch会读到\n
    // scanf("%c", &ch);

    // 正确：%c前加空格，跳过空白字符
    printf("请输入一个字符: ");
    scanf(" %c", &ch);  // 注意%c前面的空格

    printf("整数: %d, 字符: %c\n", num, ch);

    return 0;
}
安全注意事项：
  • 读取字符串时，应限制输入长度防止缓冲区溢出：scanf("%9s", str);
  • 不要使用 gets() 函数，它无法限制输入长度，已被 C11 标准移除
  • 推荐使用 fgets() 读取一整行：fgets(str, sizeof(str), stdin);

if语句

单分支if

if 语句用于根据条件执行不同的代码块，是程序实现"判断和选择"的基础。单分支 if 只有一个判断条件，满足条件时执行代码块。

单分支if示例
// 单分支if：只有"满足条件执行"
#include <stdio.h>

int main() {
    int age;

    printf("请输入年龄: ");
    scanf("%d", &age);

    if (age >= 18) {
        printf("您已成年，可以观看此内容。\n");
    }

    printf("程序结束。\n");

    return 0;
}

双分支if-else

双分支 if-else 有两个分支：条件满足执行 if 块，不满足执行 else 块。

双分支if-else示例
// 双分支if-else：满足和不满足各有一个分支
#include <stdio.h>

int main() {
    int num;

    printf("请输入一个整数: ");
    scanf("%d", &num);

    if (num % 2 == 0) {
        printf("%d是偶数\n", num);
    } else {
        printf("%d是奇数\n", num);
    }

    return 0;
}

多分支if-else if

有多个条件判断时，使用 if-else if 结构，从上到下依次检查，执行第一个满足条件的分支。

多分支if-else if示例
// 多分支if-else if：多种情况分别处理
#include <stdio.h>

int main() {
    int score;

    printf("请输入成绩: ");
    scanf("%d", &score);

    // 从上到下依次检查条件
    if (score >= 90) {
        printf("优秀！继续保持！\n");
    } else if (score >= 80) {
        printf("良好！再接再厉！\n");
    } else if (score >= 60) {
        printf("及格了，要继续努力！\n");
    } else {
        printf("不及格，需要加油！\n");
    }

    return 0;
}
执行顺序：

if-else if 结构从上到下依次检查条件，一旦找到满足的条件就执行对应代码块，然后跳过所有剩余分支。即使后面还有满足的条件也不会再执行。

条件表达式详解

if 语句的条件可以是任何能产生真（非0）或假（0）的表达式。

  • 比较运算：a > b, a == b, a != b
  • 逻辑运算：a > 0 && b < 10, a == 0 || b == 0（逻辑运算符详见第2章）
  • 直接判断：if (num) 等价于 if (num != 0)
知识点回顾：

逻辑运算符 &&（逻辑与）、||（逻辑或）、!（逻辑非）及其真值表已在第2章详细介绍，此处不再重复。在条件表达式中最常用的是 && 和 ||，用于组合多个条件。

短路特性：

&&：左边为假时，右边不执行。||：左边为真时，右边不执行。利用这个特性可以避免一些错误，例如：if (x != 0 && 10 / x > 1) 不会产生除零错误。

switch语句

switch基本语法

switch 语句适用于多分支选择，特别是当分支较多且基于同一个表达式的不同值时。它比 if-else if 更清晰、更高效。

语法格式

  • switch 表达式：必须是整数类型（int、char 等），不能是 float 或 double
  • case 标签：必须是常量表达式，不能是变量
  • break：跳出 switch 结构，防止"贯穿"执行
  • default：所有 case 都不匹配时执行，类似于 if-else if 末尾的 else
switch基本用法
// switch语句基本用法
#include <stdio.h>

int main() {
    int day;

    printf("请输入星期几(1-7): ");
    scanf("%d", &day);

    switch(day) {
        case 1:
            printf("星期一\n");
            break;
        case 2:
            printf("星期二\n");
            break;
        case 3:
            printf("星期三\n");
            break;
        case 4:
            printf("星期四\n");
            break;
        case 5:
            printf("星期五\n");
            break;
        case 6:
            printf("星期六\n");
            break;
        case 7:
            printf("星期日\n");
            break;
        default:
            printf("输入错误\n");
    }

    return 0;
}

break的作用

break 语句用于跳出 switch 结构。如果 case 分支末尾没有 break，程序会"贯穿"到下一个 case 继续执行，这通常是 bug 的来源。

缺少break的贯穿现象
// 缺少break的贯穿现象
#include <stdio.h>

int main() {
    int num = 2;

    switch(num) {
        case 1:
            printf("一\n");
            // 缺少break，会贯穿到case 2
        case 2:
            printf("二\n");
            // 缺少break，会贯穿到case 3
        case 3:
            printf("三\n");
            break;
    }
    // 输出：二  三（从case 2开始贯穿执行）

    return 0;
}
注意：

每个 case 分支末尾都应该有 break，除非你明确需要利用贯穿特性。忘记 break 是 switch 语句最常见的错误之一。

case合并与default子句

多个 case 可以执行相同的代码，将它们合并在一起。default 子句是可选的，用于处理所有 case 都不匹配的情况。

case合并示例 - 判断季节
// switch case合并 - 判断季节
#include <stdio.h>

int main() {
    int month;

    printf("请输入月份(1-12): ");
    scanf("%d", &month);

    switch(month) {
        case 12:
        case 1:
        case 2:
            printf("冬季\n");
            break;
        case 3:
        case 4:
        case 5:
            printf("春季\n");
            break;
        case 6:
        case 7:
        case 8:
            printf("夏季\n");
            break;
        case 9:
        case 10:
        case 11:
            printf("秋季\n");
            break;
        default:
            printf("无效月份\n");
    }

    return 0;
}

switch与if-else if对比

switch 和 if-else if 都可以实现多分支选择，但适用场景不同。

适合用 switch
  • 基于整数值的多分支判断
  • 分支较多（4个以上）
  • 每个条件都是相等比较
  • 代码更清晰，执行效率更高
适合用 if-else if
  • 基于范围的判断（如 score >= 90）
  • 分支较少（2-3个）
  • 需要复杂条件（逻辑运算）
  • 条件涉及浮点数比较
switch vs if-else if对比
// 使用switch的情况 - 基于整数值
int choice = 2;
switch(choice) {
    case 1: printf("选项1\n"); break;
    case 2: printf("选项2\n"); break;
    case 3: printf("选项3\n"); break;
}

// 使用if-else if的情况 - 基于范围
int score = 85;
if (score >= 90) {
    printf("优秀\n");
} else if (score >= 80) {
    printf("良好\n");
} else if (score >= 60) {
    printf("及格\n");
}
边学边练 - switch语句

练习3：编写简单计算器程序，用户输入两个数和一个运算符（+、-、*、/），使用 switch 实现四则运算。注意处理除数为0的情况。

参考答案
简单计算器
#include <stdio.h>

int main() {
    double num1, num2;
    char op;

    printf("请输入表达式(如: 5 + 3): ");
    scanf("%lf %c %lf", &num1, &op, &num2);

    switch(op) {
        case '+':
            printf("结果: %.2f\n", num1 + num2);
            break;
        case '-':
            printf("结果: %.2f\n", num1 - num2);
            break;
        case '*':
            printf("结果: %.2f\n", num1 * num2);
            break;
        case '/':
            if(num2 != 0) {
                printf("结果: %.2f\n", num1 / num2);
            } else {
                printf("错误：除数不能为0\n");
            }
            break;
        default:
            printf("错误：无效的运算符\n");
    }

    return 0;
}

        case 10: case 11: case 12:
            printf("第四季度\n"); break;
        default:
            printf("无效月份\n");
    }

    return 0;
}

选择结构嵌套

嵌套if

if 语句可以嵌套使用，实现更复杂的逻辑判断。嵌套时要注意代码的缩进和大括号的配对。

嵌套if语句示例
// 嵌套if语句示例 - 找三个数中的最大值
#include <stdio.h>

int main() {
    int a = 15, b = 20, c = 10;
    int max;

    if (a > b) {
        if (a > c) {
            max = a;
        } else {
            max = c;
        }
    } else {
        if (b > c) {
            max = b;
        } else {
            max = c;
        }
    }

    printf("三个数中的最大值是: %d\n", max);

    return 0;
}

嵌套注意事项

嵌套使用注意：
  • else 的配对原则：else 总是与最近的、未配对的 if 配对，与缩进无关
  • 大括号很重要：加上大括号可以明确代码块的范围，避免歧义
  • 避免过深嵌套：嵌套超过3层时，考虑用逻辑运算符简化或拆分函数
  • 保持缩进一致：良好的缩进让代码更易读
else配对问题示例
// else的配对问题
#include <stdio.h>

int main() {
    int x = 5, y = 3;

    // 错误的理解：else与外层if配对？
    // 实际上：else与最近的if（内层）配对
    if (x > 0)
        if (y > 10)
            printf("A\n");
    else  // 这个else与内层if配对，不是外层！
        printf("B\n");

    // 正确写法：用大括号明确配对关系
    if (x > 0) {
        if (y > 10) {
            printf("A\n");
        }
    } else {  // 这个else与外层if配对
        printf("B\n");
    }

    return 0;
}
边学边练 - 综合应用

练习5：编写 BMI 计算器，输入身高（米）和体重（公斤），计算 BMI 并判断体重状态。BMI = 体重 / (身高²)。偏瘦 <18.5，正常 18.5-24，偏胖 24-28，肥胖 >=28。

参考答案
BMI计算器
#include <stdio.h>

int main() {
    float height, weight, bmi;

    printf("请输入身高(米): ");
    scanf("%f", &height);

    printf("请输入体重(公斤): ");
    scanf("%f", &weight);

    bmi = weight / (height * height);

    printf("您的BMI值为: %.2f\n", bmi);

    if (bmi < 18.5) {
        printf("偏瘦，建议增加营养\n");
    } else if (bmi < 24) {
        printf("正常，保持健康\n");
    } else if (bmi < 28) {
        printf("偏胖，建议控制饮食\n");
    } else {
        printf("肥胖，建议减肥\n");
    }

    return 0;
}

常见错误

最容易犯的错误

初学者在学习输入输出和选择结构时，容易犯以下错误，务必注意避免。

错误1：忘记scanf的&
  • 读取整数、浮点数、字符时忘记加 &
  • 会导致程序崩溃或读取到随机值
  • 只有字符串（数组名）不需要 &
错误2：= 和 == 混淆
  • = 是赋值，== 是比较
  • if (x = 5) 是赋值，永远为真
  • 应写为 if (x == 5)
错误3：缺少大括号
  • if/else 只管一条语句，多条语句必须加 {}
  • 即使只有一条语句，也建议加 {}
  • 良好的代码习惯能减少 bug

错误代码对比

以下是常见错误的错误写法和正确写法对比：

错误1：忘记scanf的&
// 错误
int num;
scanf("%d", num);  // 缺少&，程序崩溃！

// 正确
scanf("%d", &num);
错误2：= 和 == 混淆
// 错误
if (x = 5) {  // 赋值，x变为5，条件永远为真
    printf("x等于5");
}

// 正确
if (x == 5) {  // 比较，判断x是否等于5
    printf("x等于5");
}
错误3：缺少大括号
// 错误：else只管一条语句
if (score >= 60)
    printf("及格\n");
    printf("恭喜！\n");  // 这行总会执行，不在else控制范围内
else  // 编译错误：else找不到对应的if
    printf("不及格\n");

// 正确：用大括号明确代码块
if (score >= 60) {
    printf("及格\n");
    printf("恭喜！\n");
} else {
    printf("不及格\n");
}
错误4：错误的范围判断
// 错误：数学中正确，C语言中错误
if (0 < x < 100) {  // 先算 0
    printf("在范围内");
}

// 正确：使用逻辑与运算符
if (x > 0 && x < 100) {
    printf("在范围内");
}

重点回顾

  • printf 格式化输出：掌握 %d、%f、%.nf、%c、%s、%% 等格式说明符，以及宽度对齐和精度控制
  • 转义字符：\n 换行、\t 制表符、\" 双引号、\\ 反斜杠
  • scanf 格式化输入：注意 & 取地址符的使用，%c 前加空格跳过空白字符
  • 输入缓冲区：理解缓冲区残留问题，使用 fgets 代替 gets
  • if 语句：单分支 if、双分支 if-else、多分支 if-else if
  • 逻辑运算符：&&（与）、||（或）、!（非），注意短路特性
  • switch 语句：适用于整数值多分支判断，break 防止贯穿，case 合并处理相同逻辑
  • 选择结构嵌套：注意 else 配对原则，建议始终使用大括号
  • 常见错误：忘记 &、= 和 == 混淆、缺少大括号、错误范围判断
