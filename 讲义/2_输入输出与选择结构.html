printf函数详解

printf基本用法

printf是C语言中最常用的输出函数，用于向屏幕输出格式化的文本。

什么是printf？

printf是"print formatted"的缩写，意思是"格式化打印"。它的主要作用是将数据以特定格式输出到屏幕上。printf函数定义在stdio.h头文件中，所以使用前必须包含这个头文件。

printf的基本语法

printf函数的基本格式是：printf("格式字符串", 参数列表);

  • 格式字符串：包含要输出的文本和格式说明符（如%d、%f等）
  • 参数列表：要输出的变量，按照格式说明符的顺序提供
printf基本示例
// printf基本用法示例
#include <stdio.h>

int main() {
    // 简单文本输出 - 不需要格式说明符
    printf("Hello, World!\n");
    
    // 格式化输出整数
    int age = 20;
    printf("年龄: %d\n", age);  // %d是整数的格式说明符
    
    // 格式化输出浮点数
    float height = 1.75;
    printf("身高: %.2f 米\n", height);  // %.2f保留2位小数
    
    // 格式化输出字符
    char grade = 'A';
    printf("等级: %c\n", grade);  // %c是字符的格式说明符
    
    // 一次输出多个变量
    int score = 95;
    printf("学生年龄%d岁，成绩%d分，等级%c\n", age, score, grade);
    
    return 0;
}
重要提示：

• printf的格式说明符数量必须与后面的参数数量一致
• 格式说明符的类型必须与对应参数的类型匹配
• \n表示换行，如果不加\n，下一次输出会接在同一行

scanf函数详解

scanf基本用法

scanf是C语言中最常用的输入函数，用于从键盘读取用户输入。

什么是scanf？

scanf是"scan formatted"的缩写，意思是"格式化扫描"。它从标准输入（通常是键盘）读取数据，并按照指定的格式存储到变量中。

scanf的基本语法

scanf函数的基本格式是：scanf("格式字符串", &变量1, &变量2, ...);

  • 格式字符串：指定输入数据的类型和格式
  • &变量：取地址运算符&后跟变量名，表示变量的内存地址

为什么需要&符号？

这是初学者最常遇到的问题。scanf需要知道变量在内存中的位置才能把读取的数据存进去。&运算符就是用来获取变量的内存地址的。

记忆方法：scanf要"存"数据，需要知道"存"在哪里，所以要用&获取地址。printf只是"读"数据，不需要地址。

重要提醒：

• scanf读取整数、浮点数、字符时都需要&
• 读取字符串时不需要&（因为数组名本身就是地址）
• 忘记&是最常见的错误，会导致程序崩溃

scanf基本示例
// scanf基本用法示例
#include <stdio.h>

int main() {
    int age;
    float height;
    char grade;
    char name[50];
    
    printf("请输入年龄: ");
    scanf("%d", &age);
    
    printf("请输入身高: ");
    scanf("%f", &height);
    
    printf("请输入等级: ");
    scanf(" %c", &grade);  // 注意前面的空格
    
    printf("请输入姓名: ");
    scanf("%s", name);  // 字符串不需要&
    
    printf("\n输入信息:\n");
    printf("年龄: %d\n", age);
    printf("身高: %.2f\n", height);
    printf("等级: %c\n", grade);
    printf("姓名: %s\n", name);
    
    return 0;
}

为什么需要&符号？

&是取地址运算符，scanf需要知道变量在内存中的地址才能修改变量的值。

正确与错误的对比
// 错误的写法 - 编译器会报错
#include <stdio.h>

int main() {
    int num;
    
    // 错误：缺少&
    // scanf("%d", num);  // 编译错误！
    
    // 正确：使用&
    scanf("%d", &num);
    printf("输入的数字是: %d\n", num);
    
    // 字符串数组名本身就是地址，不需要&
    char str[100];
    scanf("%s", str);  // 正确
    // scanf("%s", &str);  // 错误！
    
    return 0;
}
记忆口诀：

scanf读数据要地址，&符号不能少。字符串数组不用&，数组名就是地址。

scanf的格式控制

scanf也支持格式控制，可以限制输入的字符数。

为什么需要限制输入长度？

限制输入长度可以防止缓冲区溢出，保护程序安全。特别是在读取字符串时，如果数组只有10个元素，但用户输入了20个字符，就会造成内存溢出。

输入长度限制
// scanf格式控制示例
#include <stdio.h>

int main() {
    int num;
    char str[10];
    
    // 限制整数位数
    printf("请输入一个最多2位的数字: ");
    scanf("%2d", &num);  // 最多读2位数字
    printf("读取的数字: %d\n", num);
    
    // 限制字符串长度
    printf("请输入名字(最多9个字符): ");
    scanf("%9s", str);  // 最多读9个字符（预留1个给\0）
    printf("名字: %s\n", str);
    
    return 0;
}

scanf的常见问题

问题1：读取整数后再读字符

scanf读取整数后，输入缓冲区会残留换行符\n，导致下一个%c直接读到换行符。

解决方法：在%c前加空格：scanf(" %c", &ch);

问题2：scanf不能读取带空格的字符串

scanf遇到空格就停止读取，要读取整行需要使用gets()或fgets()函数。

if语句

if语句基本结构

if语句用于根据条件执行不同的代码块，是程序实现"判断和选择"的基础。

if语句的三种形式

1. 单分支if语句

只有一个判断条件，满足条件时执行代码块。

单分支if示例
// 单分支if：只有"满足条件执行"
#include <stdio.h>

int main() {
    int age;
    
    printf("请输入年龄: ");
    scanf("%d", &age);
    
    if (age >= 18) {
        printf("您已成年，可以观看此内容。\n");
    }
    
    printf("程序结束。\n");
    
    return 0;
}

2. 双分支if-else语句

有两个分支：条件满足执行if块，不满足执行else块。

双分支if-else示例
// 双分支if-else：满足和不满足各有一个分支
#include <stdio.h>

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    if (num % 2 == 0) {
        printf("%d是偶数\n", num);
    } else {
        printf("%d是奇数\n", num);
    }
    
    return 0;
}

3. 多分支if-else if-else语句

有多个条件判断，从上到下依次检查，执行第一个满足条件的分支。

多分支if-else if示例
// 多分支if-else if：多种情况分别处理
#include <stdio.h>

int main() {
    int score;
    
    printf("请输入成绩: ");
    scanf("%d", &score);
    
    // 从上到下依次检查条件
    if (score >= 90) {
        printf("优秀！继续保持！\n");
    } else if (score >= 80) {
        printf("良好！再接再厉！\n");
    } else if (score >= 60) {
        printf("及格了，要继续努力！\n");
    } else {
        printf("不及格，需要加油！\n");
    }
    
    return 0;
}
执行顺序：

if-else if结构从上到下依次检查条件，一旦找到满足的条件就执行对应代码块，然后跳过所有剩余分支。即使后面还有满足的条件也不会再执行。

条件表达式详解

if语句的条件可以是任何能产生真（非0）或假（0）的表达式。

  • 比较运算： a > b, a == b, a != b
  • 逻辑运算： a > 0 && b < 10, a == 0 || b == 0
  • 直接判断： if (num) 等价于 if (num != 0)

逻辑运算符

运算符 含义 示例

|| 逻辑或 if (a < 0 || a > 100)
! 逻辑非 if (!flag)

if语句的嵌套

if语句可以嵌套使用，实现更复杂的逻辑判断。

嵌套if语句示例
// 嵌套if语句示例
#include <stdio.h>

int main() {
    int age;
    char gender;
    
    printf("请输入年龄: ");
    scanf("%d", &age);
    
    printf("请输入性别(M/F): ");
    scanf(" %c", &gender);
    
    if (age >= 18) {
        printf("您已成年");
        if (gender == 'M' || gender == 'm') {
            printf("，先生可以参加工作了！\n");
        } else if (gender == 'F' || gender == 'f') {
            printf("，女士可以参加工作了！\n");
        } else {
            printf("，可以参加工作了！\n");
        }
    } else {
        printf("您还未成年");
        if (age >= 6) {
            printf("，该上小学了！\n");
        } else {
            printf("，还是幼儿园宝宝！\n");
        }
    }
    
    return 0;
}

逻辑运算符的优先级

逻辑运算符有优先级，&&的优先级高于||，但建议使用括号明确优先级。

逻辑运算符优先级示例
// 逻辑运算符优先级
#include <stdio.h>

int main() {
    int score = 85;
    int attendance = 95;
    
    // 使用括号明确优先级
    if ((score >= 60 && attendance >= 80) || score >= 90) {
        printf("及格或优秀！\n");
    }
    
    // 逻辑运算符的短路特性
    int x = 0;
    if (x != 0 && 10 / x > 1) {  // 不会除零错误
        printf("不会执行到这里\n");
    }
    
    return 0;
}
短路特性：

&&：左边为假时，右边不执行
||：左边为真时，右边不执行

常见条件判断模式

学习一些常见的条件判断模式，可以让代码更简洁。

条件判断模式示例
// 常见条件判断模式
#include <stdio.h>

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    // 1. 判断范围
    if (num >= 1 && num <= 100) {
        printf("在1-100范围内\n");
    }
    
    // 2. 判断是否为偶数
    if (num % 2 == 0) {
        printf("是偶数\n");
    } else {
        printf("是奇数\n");
    }
    
    // 3. 判断正负零
    if (num > 0) {
        printf("正数\n");
    } else if (num < 0) {
        printf("负数\n");
    } else {
        printf("零\n");
    }
    
    // 4. 多条件判断
    char grade;
    if (num >= 90) grade = 'A';
    else if (num >= 80) grade = 'B';
    else if (num >= 70) grade = 'C';
    else if (num >= 60) grade = 'D';
    else grade = 'F';
    
    printf("等级: %c\n", grade);
    
    return 0;
}

三元运算符

三元运算符语法

三元运算符是if-else的简写形式，用于简单的条件赋值。它是C语言中唯一的三元运算符（需要三个操作数）。

三元运算符的语法

变量 = 条件 ? 值1 : 值2;

  • 如果条件为真，返回值1
  • 如果条件为假，返回值2

何时使用三元运算符？

适合使用：简单的条件赋值，一行代码就能表达清楚

不适合使用：复杂的条件判断，需要执行多条语句的情况

三元运算符示例
// 三元运算符用法
#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    // 三元运算符：条件 ? 值1 : 值2
    int max = (a > b) ? a : b;
    printf("较大值: %d\n", max);
    
    // 等价的if-else（更冗长）
    int max2;
    if (a > b) {
        max2 = a;
    } else {
        max2 = b;
    }
    printf("较大值2: %d\n", max2);
    
    // 在printf中直接使用
    printf("结果: %s\n", (a > b) ? "a较大" : "b较大");
    
    // 判断奇偶数
    int num = 7;
    printf("%d是%s\n", num, (num % 2 == 0) ? "偶数" : "奇数");
    
    // 计算绝对值
    int x = -5;
    int abs_x = (x < 0) ? -x : x;
    printf("%d的绝对值是%d\n", x, abs_x);
    
    return 0;
}
使用建议：

• 三元运算符适合简单的二选一逻辑
• 如果逻辑复杂或需要多条语句，还是用if-else更清晰
• 可以嵌套使用，但不建议嵌套太多层，会降低可读性

嵌套的三元运算符

三元运算符可以嵌套，但要注意可读性。

嵌套三元运算符
// 嵌套三元运算符（不推荐过度使用）
#include <stdio.h>

int main() {
    int score = 85;
    
    // 嵌套的三元运算符
    char grade = (score >= 90) ? 'A' : 
                  (score >= 80) ? 'B' : 
                  (score >= 60) ? 'C' : 'D';
    
    printf("成绩%d，等级%c\n", score, grade);
    
    // 对比：用if-else更清晰
    char grade2;
    if (score >= 90) grade2 = 'A';
    else if (score >= 80) grade2 = 'B';
    else if (score >= 60) grade2 = 'C';
    else grade2 = 'D';
    
    return 0;
}

三元运算符练习

练习1：找最小值

使用三元运算符，编写程序输入两个数，输出较小的那个。

查看参考答案 
#include <stdio.h>

int main() {
    int a, b;
    
    printf("请输入两个整数: ");
    scanf("%d %d", &a, &b);
    
    int min = (a < b) ? a : b;
    printf("较小值是: %d\n", min);
    
    return 0;
}

练习2：限制范围

输入一个整数，如果超过100就输出100，如果小于0就输出0，否则输出原值。使用三元运算符实现。

查看参考答案 
#include <stdio.h>

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    // 嵌套三元运算符
    int result = (num > 100) ? 100 : (num < 0) ? 0 : num;
    
    printf("限制后的值: %d\n", result);
    
    return 0;
}

常见错误

最容易犯的错误

错误1：忘记scanf的& 
// 错误
int num;
scanf("%d", num);  // 缺少&

// 正确
scanf("%d", &num);
错误2：赋值与比较混淆 
// 错误
if (x = 5) {  // 赋值
    printf("相等");
}

// 正确
if (x == 5) {  // 比较
    printf("相等");
}
错误3：错误的范围判断 
// 错误
if (0 < x < 100) {  // 数学中正确，C语言中错误
    printf("在范围内");
}

// 正确
if (x > 0 && x < 100) {
    printf("在范围内");
}
