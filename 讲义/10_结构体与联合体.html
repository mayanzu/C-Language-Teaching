结构体与联合体

用自定义类型组织复杂数据，构建实用的数据结构

为什么要自定义类型？

到目前为止，我们使用的是C语言内置的基本类型：int、float、char 等。但现实世界中的数据往往是由多个不同类型的信息组合而成的。比如，一个学生的信息包括姓名（字符串）、年龄（整数）、成绩（浮点数）——这些数据需要作为一个整体来处理。

结构体（struct）就是C语言提供的"自定义类型"工具，让你可以把多个不同类型的数据打包在一起，形成一个新的类型。

新手注意：
  • 结构体就像一个"信息卡片"，可以包含不同类型的字段
  • 联合体（union）看起来和结构体很像，但它在同一时刻只能存储一个成员的值
  • 枚举（enum）用于定义一组有名字的整数常量
生活类比：

基本类型像单间，只能放一种东西；结构体像套房，有客厅、卧室、厨房，每个房间可以放不同类型的东西；联合体像变形房间，同一时间只能是客厅或卧室或厨房。

结构体定义与使用

定义结构体

使用 struct 关键字定义结构体类型，在大括号内列出所有成员变量。

定义学生结构体
// 定义结构体类型
struct Student {
    char name[20];   // 姓名
    int age;           // 年龄
    float score;       // 成绩
};

// 注意：定义结构体类型后，
// 还需要声明变量才能使用
struct Student stu1;    // 声明一个Student类型的变量
struct Student stu2;
要点：
  • struct Student 是类型名，stu1 是变量名
  • 结构体定义末尾的 ; 不能忘记
  • 定义结构体类型不会分配内存，声明变量时才分配

结构体的初始化

结构体初始化方式
#include <stdio.h>
#include <string.h>

struct Student {
    char name[20];
    int age;
    float score;
};

int main() {
    // 方式1：定义时按顺序初始化
    struct Student stu1 = {"张三", 20, 89.5};

    // 方式2：指定成员初始化（推荐，不怕顺序错）
    struct Student stu2 = {.name = "李四", .age = 21, .score = 92.0};

    // 方式3：先声明再赋值
    struct Student stu3;
    strcpy(stu3.name, "王五");
    stu3.age = 19;
    stu3.score = 85.5;

    printf("%s, %d岁, 成绩%.1f\n", stu1.name, stu1.age, stu1.score);

    return 0;
}

结构体成员访问

点运算符（.）和箭头运算符（->）

访问结构体成员有两种方式，取决于你用的是结构体变量还是结构体指针。

运算符 使用场景 示例 记忆方法
.（点） 通过结构体变量访问 stu.age 变量是"实体"，用实心点
->（箭头） 通过结构体指针访问 p->age 指针是"方向"，用箭头
两种访问方式对比
#include <stdio.h>

struct Point {
    int x;
    int y;
};

int main() {
    struct Point p1 = {3, 4};
    struct Point *p2 = &p1;

    // 通过变量访问：用 .
    printf("p1.x = %d, p1.y = %d\n", p1.x, p1.y);

    // 通过指针访问：用 ->
    printf("p2->x = %d, p2->y = %d\n", p2->x, p2->y);

    // p2->x 等价于 (*p2).x
    printf("(*p2).x = %d\n", (*p2).x);

    return 0;
}
等价关系：

p->member 等价于 (*p).member。箭头运算符是简化写法，更直观也更常用。

结构体嵌套

结构体的成员也可以是另一个结构体，这就是嵌套。

结构体嵌套示例
#include <stdio.h>

struct Date {
    int year;
    int month;
    int day;
};

struct Employee {
    char name[20];
    struct Date birthday;   // 嵌套结构体
    float salary;
};

int main() {
    struct Employee emp = {
        .name = "张三",
        .birthday = {2000, 5, 15},
        .salary = 8000.0
    };

    // 访问嵌套成员：用多个 .
    printf("%s的生日: %d-%d-%d\n",
        emp.name,
        emp.birthday.year,
        emp.birthday.month,
        emp.birthday.day);

    return 0;
}

结构体数组

定义与使用结构体数组

当需要处理多个同类型的结构体数据时，可以使用结构体数组。就像普通数组存储多个 int 一样，结构体数组存储多个结构体变量。

学生信息管理
#include <stdio.h>

struct Student {
    char name[20];
    int age;
    float score;
};

int main() {
    // 定义并初始化结构体数组
    struct Student class[] = {
        {"张三", 20, 89.5},
        {"李四", 21, 92.0},
        {"王五", 19, 78.5},
        {"赵六", 20, 95.0}
    };

    int n = sizeof(class) / sizeof(class[0]);

    // 遍历输出所有学生信息
    printf("姓名\t年龄\t成绩\n");
    for (int i = 0; i < n; i++) {
        printf("%s\t%d\t%.1f\n", class[i].name, class[i].age, class[i].score);
    }

    return 0;
}

结构体数组排序

按成绩从高到低排序
#include <stdio.h>
#include <string.h>

struct Student {
    char name[20];
    int age;
    float score;
};

int main() {
    struct Student class[] = {
        {"张三", 20, 89.5},
        {"李四", 21, 92.0},
        {"王五", 19, 78.5},
        {"赵六", 20, 95.0}
    };
    int n = sizeof(class) / sizeof(class[0]);

    // 冒泡排序：按成绩降序
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1 - i; j++) {
            if (class[j].score < class[j + 1].score) {
                // 交换整个结构体
                struct Student temp = class[j];
                class[j] = class[j + 1];
                class[j + 1] = temp;
            }
        }
    }

    printf("成绩排名:\n");
    for (int i = 0; i < n; i++) {
        printf("第%d名: %s %.1f分\n", i + 1, class[i].name, class[i].score);
    }

    return 0;
}
提示：

结构体变量可以直接赋值（temp = class[j]），C语言会自动复制所有成员。这比逐个成员复制方便多了。

结构体指针

结构体指针的基本用法

结构体指针就是指向结构体变量的指针。通过结构体指针，我们可以高效地操作结构体数据，避免复制整个结构体的开销。

结构体指针
#include <stdio.h>

struct Student {
    char name[20];
    int age;
    float score;
};

int main() {
    struct Student stu = {"张三", 20, 89.5};
    struct Student *p = &stu;

    // 通过指针访问成员（两种方式等价）
    printf("姓名: %s\n", p->name);      // 推荐用 ->
    printf("年龄: %d\n", (*p).age);    // 等价写法
    printf("成绩: %.1f\n", p->score);

    // 通过指针修改成员
    p->age = 21;
    printf("修改后年龄: %d\n", stu.age);  // 输出: 21

    return 0;
}

结构体指针与动态内存

动态分配结构体
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Student {
    char name[20];
    int age;
    float score;
};

int main() {
    // 动态分配一个结构体
    struct Student *p = (struct Student *)malloc(sizeof(struct Student));

    if (p == NULL) {
        printf("内存分配失败！\n");
        return 1;
    }

    strcpy(p->name, "张三");
    p->age = 20;
    p->score = 89.5;

    printf("%s, %d岁, %.1f分\n", p->name, p->age, p->score);

    free(p);
    p = NULL;

    return 0;
}
边学边练 - 结构体指针

练习2： 编写函数，使用结构体指针作为参数，实现计算一组学生平均成绩的功能。

参考答案
计算平均成绩
#include <stdio.h>

typedef struct {
    char name[20];
    float score;
} Student;

float average(const Student *students, int n) {
    float sum = 0;
    for (int i = 0; i < n; i++) {
        sum += (students + i)->score;
    }
    return sum / n;
}

int main() {
    Student class[] = {
        {"张三", 89.5},
        {"李四", 92.0},
        {"王五", 78.5}
    };
    int n = sizeof(class) / sizeof(class[0]);

    printf("平均成绩: %.2f\n", average(class, n));

    return 0;
}

结构体与函数

结构体作为函数参数

结构体传给函数有三种方式：值传递、指针传递、结构体指针作为返回值。

三种传递方式对比
#include <stdio.h>

struct Point {
    int x;
    int y;
};

// 方式1：值传递 — 复制整个结构体（效率低）
void print_point_value(struct Point p) {
    printf("(%d, %d)\n", p.x, p.y);
}

// 方式2：指针传递 — 只传地址（效率高，推荐）
void print_point_ptr(const struct Point *p) {
    printf("(%d, %d)\n", p->x, p->y);
}

// 方式3：通过指针修改结构体
void move_point(struct Point *p, int dx, int dy) {
    p->x += dx;
    p->y += dy;
}

int main() {
    struct Point p = {3, 4};

    print_point_value(p);    // 值传递
    print_point_ptr(&p);     // 指针传递

    move_point(&p, 10, 20);
    printf("移动后: (%d, %d)\n", p.x, p.y);  // 输出: (13, 24)

    return 0;
}
最佳实践：
  • 如果只是读取数据，用 const struct Point *p，既高效又安全
  • 如果需要修改数据，用 struct Point *p
  • 尽量避免值传递，结构体可能很大，复制开销高

结构体内存对齐

sizeof 的"意外"结果

你可能会发现，sizeof 计算结构体大小时，结果往往大于所有成员大小之和。这是因为编译器进行了内存对齐。

内存对齐示例
#include <stdio.h>

struct A {
    char c;    // 1字节
    int i;     // 4字节
};

struct B {
    int i;     // 4字节
    char c;    // 1字节
};

int main() {
    printf("struct A: %d字节\n", sizeof(struct A));  // 8字节（不是5！）
    printf("struct B: %d字节\n", sizeof(struct B));  // 8字节
    return 0;
}
对齐规则（简化版）：
  • 每个成员的起始地址必须是自身大小的整数倍（如 int 起始地址必须是4的倍数）
  • 结构体总大小必须是最大成员大小的整数倍
  • 编译器会在成员之间插入"填充字节"来满足对齐要求
为什么要对齐？

CPU访问对齐的内存更快。虽然浪费了一点空间，但换取了更高的运行速度。这是典型的"空间换时间"策略。

节省空间的技巧

调整成员的声明顺序，可以减少填充字节，节省内存：

成员排列优化
// 浪费空间：char后面填充3字节
struct Bad {
    char a;    // 1 + 3字节填充
    int b;     // 4
    char c;    // 1 + 3字节填充
};  // 总共12字节

// 省空间：按大小降序排列
struct Good {
    int b;     // 4
    char a;    // 1
    char c;    // 1 + 2字节填充
};  // 总共8字节
建议：结构体成员按大小从大到小排列（double → int → short → char），可以最大程度减少填充字节。

链表基础

什么是链表？

链表是一种动态数据结构，通过结构体指针将多个节点串联起来。每个节点包含数据和一个指向下一个节点的指针。

类比：火车 vs 链表
  • 数组 = 一排连在一起的火车车厢（固定、连续）
  • 链表 = 用绳索连接的独立车厢（灵活、可增删）
  • 数组大小固定，链表可以随时增加或删除节点

定义链表节点

链表节点定义
#include <stdio.h>
#include <stdlib.h>

// 定义链表节点
typedef struct Node {
    int data;              // 存储数据
    struct Node *next;  // 指向下一个节点的指针
} Node;
理解 struct Node *next：

为什么节点内要写 struct Node 而不是 Node？因为此时 typedef 还没完成，新名字 Node 还不能使用。这是C语言的一个特殊规则——自引用结构体必须用完整的 struct 名称。

创建和遍历链表

手动创建3个节点的链表
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node *next;
} Node;

int main() {
    // 创建3个节点
    Node *n1 = (Node *)malloc(sizeof(Node));
    Node *n2 = (Node *)malloc(sizeof(Node));
    Node *n3 = (Node *)malloc(sizeof(Node));

    // 设置数据和连接
    n1->data = 10;  n1->next = n2;
    n2->data = 20;  n2->next = n3;
    n3->data = 30;  n3->next = NULL;  // 最后一个节点的next为NULL

    // 遍历链表
    Node *p = n1;  // 从头节点开始
    while (p != NULL) {
        printf("%d -> ", p->data);
        p = p->next;  // 移动到下一个节点
    }
    printf("NULL\n");
    // 输出: 10 -> 20 -> 30 -> NULL

    // 释放内存
    free(n1); free(n2); free(n3);

    return 0;
}
关键要点：
  • 链表的最后一个节点，next 必须为 NULL，表示"后面没有了"
  • 遍历时用 p = p->next 走到下一个节点
  • 遍历条件 p != NULL 确保不会访问空指针
  • 链表节点用 malloc 动态分配，用完必须 free

联合体

联合体的概念

联合体（union）的声明方式与结构体类似，但有一个关键区别：联合体的所有成员共享同一块内存，同一时刻只能存储一个成员的值。

联合体基本用法
#include <stdio.h>

union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    union Data data;

    printf("sizeof(union Data) = %lu\n", sizeof(data));
    // 输出: 20（最大成员 str[20] 的大小）

    // 只能使用一个成员
    data.i = 10;
    printf("data.i = %d\n", data.i);  // 输出: 10

    data.f = 3.14;
    printf("data.f = %.2f\n", data.f);  // 输出: 3.14
    // 此时 data.i 的值已被覆盖！

    return 0;
}
重要：

联合体中写入一个成员后，其他成员的值会变得不确定。必须确保读取的是最后写入的成员，否则会得到无意义的数据。

struct 与 union 对比

特性 struct（结构体） union（联合体）
内存分配 各成员各自占用空间 所有成员共享空间
总大小 所有成员大小之和（考虑对齐） 最大成员的大小
同时存储 可以同时存储所有成员 同一时刻只能存一个成员
典型用途 组合不同类型的数据 节省内存、类型转换

练习3： 定义一个枚举表示季节（春、夏、秋、冬），编写函数根据月份返回对应的季节。

参考答案
月份判断季节
#include <stdio.h>

typedef enum { SPRING, SUMMER, AUTUMN, WINTER } Season;

Season get_season(int month) {
    if (month >= 3 && month <= 5)  return SPRING;
    if (month >= 6 && month <= 8)  return SUMMER;
    if (month >= 9 && month <= 11) return AUTUMN;
    return WINTER;
}

const char *season_name(Season s) {
    static const char *names[] = {"春天", "夏天", "秋天", "冬天"};
    return names[s];
}

int main() {
    int month;
    printf("输入月份(1-12): ");
    scanf("%d", &month);

    Season s = get_season(month);
    printf("%d月是%s\n", month, season_name(s));

    return 0;
}

枚举类型

枚举的定义与使用

枚举（enum）用于定义一组有名字的整数常量，让代码更易读。比如用 RED 代替 0，用 GREEN 代替 1。

枚举类型示例
#include <stdio.h>

// 定义枚举类型
enum Color { RED, GREEN, BLUE };          // 值: 0, 1, 2
enum Weekday { MON=1, TUE, WED, THU, FRI, SAT, SUN };  // 值: 1~7

int main() {
    enum Color c = GREEN;
    printf("颜色值: %d\n", c);  // 输出: 颜色值: 1

    enum Weekday today = WED;
    printf("星期三 = %d\n", today);  // 输出: 星期三 = 3

    // 用枚举让代码更易读
    if (today == SAT || today == SUN) {
        printf("周末！\n");
    } else {
        printf("工作日\n");
    }

    return 0;
}
枚举的要点：
  • 默认从0开始，每个后续值加1
  • 可以指定起始值：MON=1，后续自动递增
  • 枚举值本质是 int，可以参与整数运算
  • 使用枚举代替"魔数"（如 0、1），代码更可读

typedef

用typedef简化类型名

typedef 用于给已有的类型起一个新名字（别名），让代码更简洁易读。

typedef 使用示例
#include <stdio.h>

// 给结构体起别名
typedef struct {
    char name[20];
    int age;
    float score;
} Student;

// 给枚举起别名
typedef enum { RED, GREEN, BLUE } Color;

// 给指针类型起别名
typedef int *IntPtr;

int main() {
    // 不用再写 struct 关键字！
    Student stu = {"张三", 20, 89.5};
    Color c = GREEN;
    IntPtr p = &stu.age;

    printf("%s, %d岁, %.1f分\n", stu.name, stu.age, stu.score);
    printf("颜色: %d\n", c);
    printf("年龄: %d\n", *p);

    return 0;
}
typedef 的好处
  • 简化冗长的类型名
  • 提高代码可读性
  • 方便跨平台移植
  • 省去 struct 关键字
常见用法
  • typedef struct {...} Name;
  • typedef int BOOL;
  • typedef void (*FuncPtr)(int);
边学边练 - typedef

练习4： 使用联合体实现一个"万能数据容器"，可以存储 int、float 或字符串，并用枚举记录当前存储的数据类型。

参考答案
万能数据容器
#include <stdio.h>
#include <string.h>

typedef enum { TYPE_INT, TYPE_FLOAT, TYPE_STRING } DataType;

typedef struct {
    DataType type;
    union {
        int i_val;
        float f_val;
        char str[50];
    } data;
} Value;

void print_value(const Value *v) {
    switch (v->type) {
        case TYPE_INT:    printf("int: %d\n", v->data.i_val); break;
        case TYPE_FLOAT:  printf("float: %.2f\n", v->data.f_val); break;
        case TYPE_STRING: printf("string: %s\n", v->data.str); break;
    }
}

int main() {
    Value v1, v2, v3;

    v1.type = TYPE_INT;
    v1.data.i_val = 42;

    v2.type = TYPE_FLOAT;
    v2.data.f_val = 3.14;

    v3.type = TYPE_STRING;
    strcpy(v3.data.str, "Hello");

    print_value(&v1);
    print_value(&v2);
    print_value(&v3);

    return 0;
}

练习1： 定义一个学生结构体（学号、姓名、三门课成绩），写程序输入5个学生信息，计算每个学生的平均分并输出最高平均分的学生信息。

#include <stdio.h>

struct Student {
    int id;
    char name[20];
    float scores[3];
    float average;
};

int main() {
    struct Student stu[5];

    for (int i = 0; i < 5; i++) {
        scanf("%d %s %f %f %f", &stu[i].id, stu[i].name,
              &stu[i].scores[0], &stu[i].scores[1], &stu[i].scores[2]);
        stu[i].average = (stu[i].scores[0] + stu[i].scores[1] + stu[i].scores[2]) / 3;

    for (int i = 1; i < 5; i++)
        if (stu[i].average > stu[max_idx].average) max_idx = i;

    printf("最高平均分: %s, 平均: %.1f\n", stu[max_idx].name, stu[max_idx].average);

}

练习2： 使用结构体指针和动态内存分配，创建一个链表并实现插入、遍历功能。

参考答案 
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node *next;
};

// 头插法
struct Node* insert(struct Node *head, int val) {
    struct Node *newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = val;
    newNode->next = head;
    return newNode;
}

void printList(struct Node *head) {
    while (head) {
        printf("%d -> ", head->data);
        head = head->next;
    }
    printf("NULL\n");
}

int main() {
    struct Node *head = 0;
    head = insert(head, 3);
    head = insert(head, 2);
    head = insert(head, 1);
    printList(head);  // 1 -> 2 -> 3 -> NULL
    return 0;
}

练习5： 解释结构体和联合体的主要区别，以及各自的使用场景。

参考答案
  • 结构体：所有成员同时存在，各自占独立内存，总大小≥各成员大小之和。场景：描述一个实体的多个属性（如学生有姓名、年龄、成绩）。
  • 联合体：所有成员共享同一段内存，同一时刻只能使用一个成员，大小=最大成员的大小。场景：同一数据有多种解读方式（如IP地址可以是整数也可以是4个字节）。

练习6： 写一个程序，使用结构体存储日期（年、月、日），写一个函数判断两个日期的先后顺序。

参考答案 
#include <stdio.h>

typedef struct {
    int year, month, day;
} Date;

// 返回 -1: d1<d2, 0: 相等, 1: d1>d2
int compareDate(Date d1, Date d2) {
    if (d1.year != d2.year) return d1.year > d2.year ? 1 : -1;
    if (d1.month != d2.month) return d1.month > d2.month ? 1 : -1;
    if (d1.day != d2.day) return d1.day > d2.day ? 1 : -1;
    return 0;
}

int main() {
    Date d1 = {2024, 3, 15}, d2 = {2024, 5, 1};
    int r = compareDate(d1, d2);
    printf("%s\n", r < 0 ? "d1更早" : r > 0 ? "d1更晚" : "相同日期");
    return 0;
}

本章小结

重点回顾

  • 结构体定义：用 struct 将多个不同类型的数据组合在一起
  • 成员访问：变量用 .，指针用 ->
  • 结构体数组：存储多个同类型结构体，可排序
  • 结构体指针：高效传递，避免复制整个结构体
  • 内存对齐：编译器会在成员间插入填充字节，按大小降序排列可节省空间
  • 链表：通过结构体指针自引用，实现动态数据结构
  • 联合体：所有成员共享内存，同一时刻只存一个值
  • 枚举：定义有名字的整数常量，提高代码可读性
  • typedef：给类型起别名，简化代码
  • 结构体与函数：传值不影响原结构体，传指针可修改且高效
  • 内存对齐：结构体大小通常大于各成员大小之和，用sizeof获取实际大小
访问方式速查
  • 结构体变量：stu.name
  • 结构体指针：p->name
  • 等价关系：p->name ≡ (*p).name
自定义类型对比
  • struct：各成员独立存储
  • union：共享内存，省空间
  • enum：命名常量，更可读
