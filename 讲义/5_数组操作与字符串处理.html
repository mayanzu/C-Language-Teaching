数组操作与字符串处理

掌握 C 语言数组和字符串的基本操作方法

数组与字符串基础

数组是C语言中最常用的数据结构之一，用于存储相同类型的数据集合。字符串是特殊的字符数组，以空字符('\0')结尾。

新手注意：
  • 数组下标从0开始
  • 字符串必须以'\0'结尾
  • 数组名就是数组首元素的地址

数组基础

一维数组定义与初始化

什么是数组？

数组是C语言中用来存储一组相同类型数据的数据结构。可以把数组想象成一排连续的储物柜，每个柜子都有编号（下标），用来存放相同类型的物品。

数组的基本特点

  • 类型统一：数组中的所有元素必须是同一种数据类型（如全是int或全是double）
  • 连续存储：数组元素在内存中是连续排列的，相邻元素地址紧挨着
  • 下标访问：通过下标（索引）访问元素，下标从0开始，不是从1开始
  • 固定大小：数组一旦定义，大小就固定了，不能动态改变

数组定义的三种方式

数组定义示例
// C语言数组定义与初始化示例
#include <stdio.h>

int main() {
    int i;
    // 方式1：只声明不初始化（不推荐，元素值不确定）
    int a[5];  // 定义了5个int元素的数组，但未初始化
    
    // 方式2：部分初始化（推荐，未赋值的元素自动初始化为0）
    int b[5] = {1, 2, 3};  // b[0]=1, b[1]=2, b[2]=3, b[3]=0, b[4]=0
    
    // 方式3：完全初始化（推荐，可省略数组大小）
    int c[] = {1, 2, 3, 4, 5};  // 编译器自动计算大小为5
    
    // 全部初始化为0（常用技巧）
    int d[5] = {0};  // 所有元素都初始化为0
    
    // 访问数组元素（下标从0开始）
    b[3] = 10;  // 修改第4个元素（下标3）
    printf("b[3] = %d\n", b[3]);  // 输出：b[3] = 10
    
    // 遍历数组
    printf(for (i =an>(int i = 0; i < 5; i++) {
        printf("%d ", c[i]);
    }
    printf("\n");
    
    return 0;
}
⚠️ 常见错误：
  • 下标越界：访问a[5]时，如果数组只有5个元素（a[0]到a[4]），就会越界
  • 忘记初始化：未初始化的数组元素包含垃圾值
  • 数组大小必须是常量：不能用变量定义数组大小（在C99之前）

字符串基础

字符数组与字符串

什么是字符串？

在C语言中，字符串并不是一个独立的数据类型，而是字符数组的一种特殊形式。字符串的本质就是一个以'\0'（空字符）结尾的字符数组。

字符数组 vs 字符串

  • 字符数组：普通的char类型数组，可以存储任意字符，不一定以'\0'结尾
  • 字符串：以'\0'结尾的字符数组，'\0'是字符串结束的标志（ASCII码为0）
  • 关键区别：有'\0'才能被当作字符串处理，才能用%s输出

字符串的三种定义方式

字符串定义示例
// 字符串定义与使用
#include <stdio.h>
#include <string.h>

int main() {
    // 方式1：字符串字面量初始化（推荐）
    char str1[] = "Hello";  // 编译器自动添加'\0'，实际长度为6
    
    // 方式2：指定数组大小
    char str2[10] = "World";  // 数组大小为10，"World"占6个字符（含'\0'）
    
    // 方式3：字符逐个初始化（必须手动添加'\0'）
    char str3[] = {'H', 'i', '\0'};  // 忘记'\0'就不是字符串了！
    
    // 错误示例：没有'\0'的字符数组
    char notStr[] = {'A', 'B', 'C'};  // 不是字符串！用%s输出会出错
    
    // 输出字符串
    printf("str1: %s (长度: %u)\n", str1, strlen(str1));
    printf("str2: %s (长度: %u)\n", str2, strlen(str2));
    printf("str3: %s (长度: %u)\n", str3, strlen(str3));
    
    // 访问单个字符
    printf("str1的第一个字符: %c\n", str1[0]);  // 'H'
    printf("str1的结束符: %d\n", str1[5]);    // 0（'\0'的ASCII码）
    
    // 修改字符串中的字符
    str1[0] = 'h';
    printf("修改后的str1: %s\n", str1);  // "hello"
    
    return 0;
}
⚠️ 重要提示：
  • strlen()返回的是字符串中有效字符的个数，不包括'\0'
  • sizeof()返回的是数组占用的总字节数，包括'\0'
  • 字符串常量（如"Hello"）存储在只读内存区，不能修改
  • 字符串数组可以修改单个字符，但不能整体赋值（str1 = "new"; 是错误的）

字符串函数详解

函数 功能 示例 返回值
strlen(str) 计算字符串长度 strlen("Hello") 5 (不含'\0')
strcpy(dest, src) 字符串复制 strcpy(a, b) dest指针
strcat(dest, src) 字符串连接 strcat(a, b) dest指针
strcmp(str1, str2) 字符串比较 strcmp("abc", "abd") <0, =0, >0
strncpy(dest, src, n) 复制n个字符 strncpy(a, b, 3) dest指针
strncmp(str1, str2, n) 比较前n个字符 strncmp(a, b, 3) <0, =0, >0

#include <stdio.h>
#include <string.h>
#include <ctype.h>

int checkPasswordStrength(const char *password) {
    int length = strlen(password);
    int has_upper = 0, has_lower = 0, has_digit = 0, has_special = 0;
    
    if (length < 8) return 0;  // 太短
    
    for (i = 0; i < length; i++) {
        if (isupper(password[i])) has_upper = 1;
        else if (islower(password[i])) has_lower = 1;
        else if (isdigit(password[i])) has_digit = 1;
        else has_special = 1;
    }
    
    int score = has_upper + has_lower + has_digit + has_special;
    return score;
}

// 选择排序和插入排序
#include <stdio.h>

// 选择排序
void selectionSort(int arr[], int n) {
    for (i = 0; i < n-1; i++) {
        int min_idx = i;
        for (j = i+1; j < n; j++) {
            if(arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        // 交换
        int temp = arr[min_idx];
        arr[min_idx] = arr[i];
        arr[i] = temp;
    }
}

// 插入排序
void insertionSort(int arr[], int n) {
    for (i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;
        
        while(j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key;
    }
}

void printArray(int arr[], int n) {
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int main() {
    int arr1[] = {64, 25, 12, 22, 11};
    int arr2[] = {12, 11, 13, 5, 6};
    int n1 = sizeof(arr1)/sizeof(arr1[0]);
    int n2 = sizeof(arr2)/sizeof(arr2[0]);
    
    printf("选择排序:\n");
    printf("原数组: ");
    printArray(arr1, n1);
    selectionSort(arr1, n1);
    printf("排序后: ");
    printArray(arr1, n1);
    
    printf("\n插入排序:\n");

    printArray(arr2, n2);

数组查找算法

二分查找
// 二分查找（要求数组已排序）
#include <stdio.h>

int binarySearch(int arr[], int left, int right, int target) {
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        if (arr[mid] == target) {
            return mid;  // 找到目标
        }
        
        if (arr[mid] < target) {
            left = mid + 1;  // 在右半部分查找
        } else {
            right = mid - 1;  // 在左半部分查找
        }
    }
    
    return -1;  // 没有找到
}

int main() {
    int i;
    int arr[] = {2, 3, 4, 10, 40, 50, 60, 70};
    int n = sizeof(arr)/sizeof(arr[0]);
    int target;
    
    printf(for (i =an>(int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    
    printf("请输入要查找的数字: ");
    scanf("%d", &target);
    
    int result = binarySearch(arr, 0, n-1, target);
    
    if (result != -1) {
        printf("元素 %d 找到在位置 %d\n", target, result);

    return 0;
}

#include <stdio.h>

int binarySearch(int arr[], int left, int right, int target) {
    while (left <= right) {
        int mid = left + (right - left)/2;
        if (arr[mid] == target) return mid;
        if (arr[mid] < target) left = mid + 1; else right = mid - 1;
    }
    return -1;
}

int main() {
