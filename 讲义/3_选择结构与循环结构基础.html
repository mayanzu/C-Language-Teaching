选择结构与循环结构基础

掌握 C 语言条件判断和循环控制的基本方法

选择结构与循环结构基础

C语言程序需要根据不同条件执行不同操作，这就是选择结构。同时，程序经常需要重复执行某些操作，这就是循环结构。

新手注意：
  • 选择结构让程序能够做出判断和选择
  • 循环结构让程序能够重复执行任务
  • 掌握break和continue语句很重要
重要提示：选择结构和循环结构是C语言的核心控制结构，理解它们的执行流程是编程的基础。

学习目标与节奏

按照下面的节奏来学习，确保每一步都扎实：

  1. 第1步：掌握if多分支 —— 学会用if-else if处理多种情况
  2. 第2步：理解switch语句 —— 掌握多分支选择的另一种写法
  3. 第3步：练习嵌套结构 —— 理解选择结构的嵌套使用
  4. 第4步：入门循环控制 —— 掌握while、for、do-while三种循环

if多分支选择

if-else if结构

当需要处理多种互斥的情况时，使用if-else if结构。

成绩等级判断
// if-else if多分支选择示例
#include <stdio.h>

int main() {
    int score;
    
    printf("请输入成绩: ");
    scanf("%d", &score);
    
    if (score >= 90) {
        printf("优秀\n");
    } else if (score >= 80) {
        printf("良好\n");
    } else if (score >= 60) {
        printf("及格\n");
    } else {
        printf("不及格\n");
    }
    
    return 0;
}

if-else if的执行流程

if-else if结构从上到下依次检查条件，找到第一个为真的条件就执行对应的代码块，然后跳出整个结构。

执行流程演示
// if-else if执行流程
#include <stdio.h>

int main() {
    int score = 85;
    
    printf("成绩: %d\n", score);
    
    if (score >= 90) {
        printf("等级: 优秀\n");
        printf("奖励: 奖金100元\n");
    } else if (score >= 80) {
        printf("等级: 良好\n");
        printf("奖励: 奖金50元\n");
    } else if (score >= 70) {
        printf("等级: 中等\n");
        printf("奖励: 鼓励奖\n");
    } else if (score >= 60) {
        printf("等级: 及格\n");
        printf("奖励: 继续努力\n");
    } else {
        printf("等级: 不及格\n");
        printf("奖励: 需要补考\n");
    }
    
    return 0;
}
执行顺序：

1. 从上到下依次检查条件
2. 遇到第一个为真的条件，执行对应代码块
3. 跳过剩余的所有分支
4. 如果所有条件都为假，执行else块

常见的多分支模式

学习一些常见的多分支判断模式，可以让代码更简洁。

多分支判断模式
// 常见的多分支判断模式
#include <stdio.h>

int main() {
    int num;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    // 1. 判断正负零
    if (num > 0) {
        printf("正数\n");
    } else if (num < 0) {
        printf("负数\n");
    } else {
        printf("零\n");
    }
    
    // 2. 判断范围区间
    if (num >= 0 && num <= 100) {
        printf("在0-100范围内\n");
    } else if (num >= 101 && num <= 200) {
        printf("在101-200范围内\n");
    } else {
        printf("超出范围\n");
    }
    
    // 3. 成绩分段
    char grade;
    if (num >= 90) grade = 'A';
    else if (num >= 80) grade = 'B';
    else if (num >= 70) grade = 'C';
    else if (num >= 60) grade = 'D';
    else grade = 'F';
    
    printf("等级: %c\n", grade);
    
    return 0;
}

switch vs if-else if

switch和if-else if都可以实现多分支选择，什么时候用哪个？

情况 推荐使用 理由
基于整数值的多分支 switch 代码更清晰，执行效率更高
基于范围的判断 if-else if switch不支持范围判断
分支较少(2-3个) if-else if switch结构相对复杂
需要复杂条件 if-else if switch只能做相等比较

    case 2: printf("选项2\n"); break;

}

// 使用if-else if的情况
int score = 85;
if (score >= 90) {
    printf("优秀\n");
} else if (score >= 80) {
    printf("良好\n");
} else if (score >= 60) {

while循环

while循环语法

while循环在每次执行循环体之前检查条件。

while循环示例
// while循环基本用法
#include <stdio.h>

int main() {
    int i = 1;
    
    printf("while循环:\n");
    while (i <= 5) {
        printf("第%d次循环\n", i);
        i++;  // 不要忘记更新循环变量
    }
    
    return 0;
}

while循环的执行流程

while循环的执行顺序：先判断条件，再执行循环体。

while循环执行流程
// while循环执行流程演示
#include <stdio.h>

int main() {
    int i = 1;
    
    printf("while循环开始:\n");
    while (i <= 3) {
        printf("  第%d次循环开始，i=%d\n", i, i);
        printf("  执行循环体\n");
        i++;  // 更新循环变量
        printf("  第%d次循环结束，i=%d\n", i-1, i);
    }
    printf("while循环结束\n");
    
    return 0;
}
while循环特点：

• 先判断条件，后执行循环体
• 如果条件一开始就为假，循环体一次都不执行
• 循环体内必须更新循环变量，否则可能死循环

while循环的常见应用

while循环常用于不确定循环次数的情况。

while循环应用示例
// while循环的常见应用
#include <stdio.h>

int main() {
    int sum = 0, num;
    
    printf("输入数字求和(输入0结束):\n");
    while (1) {  // 无限循环
        printf("请输入数字: ");
        scanf("%d", &num);
        
        if (num == 0) {
            break;  // 输入0时退出
        }
        
        sum += num;
        printf("当前和: %d\n", sum);
    }
    
    printf("最终和: %d\n", sum);
    
    // 猜数字游戏
    int target = 42, guess;
    int attempts = 0;
    
    printf("\n猜数字游戏(1-100):\n");
    while (1) {
        printf("请输入猜测: ");
        scanf("%d", &guess);
        attempts++;
        
        if (guess == target) {
            printf("恭喜猜对了! 用了%d次\n", attempts);
            break;
        } else if (guess < target) {
            printf("太小了!\n");
        } else {
            printf("太大了!\n");
        }
    }
    
    return 0;
}

练习2：数字反转

输入一个整数，使用while循环将其各位数字反转后输出。例如：输入1234，输出4321。

查看参考答案 
#include <stdio.h>

int main() {
    int num, reversed = 0, digit;
    
    printf("请输入一个整数: ");
    scanf("%d", &num);
    
    int original = num;  // 保存原始值
    
    while (num != 0) {
        digit = num % 10;           // 取最后一位
        reversed = reversed * 10 + digit;  // 构建反转数
        num /= 10;                 // 去掉最后一位
    }
    
    printf("%d 反转后是 %d\n", original, reversed);
    
    return 0;
}

for循环的执行流程

for循环的执行顺序：初始化 → 判断条件 → 执行循环体 → 更新 → 再次判断条件...

        isPrime = 1;  // 假设是素数

}

do-while循环

do-while vs while的区别

do-while至少执行一次循环体，而while可能一次都不执行。

do-while vs while对比
// do-while vs while的区别
#include <stdio.h>

int main() {
    int x = 10;
    
    printf("x = %d\n", x);
    printf("while循环:\n");
    
    while (x < 5) {  // 条件为假，不执行
        printf("  while: x = %d\n", x);
        x++;
    }
    
    printf("do-while循环:\n");
    x = 10;  // 重置x
    do {
        printf("  do-while: x = %d\n", x);
        x++;
    } while (x < 5);  // 条件为假，但已执行了一次
    
    printf("循环都结束了!\n");
    
    return 0;
}
选择建议：

• 需要至少执行一次：用do-while
• 可能一次都不执行：用while
• 已知循环次数：用for

三种循环的比较

练习：至少执行一次 vs 可能不执行

读取用户输入的年龄（整数）。如果输入为负数，while循环不会执行；do-while循环则至少提示一次，直到输入合法年龄（0-150）。

查看参考答案 
// while 与 do-while 对比示例
#include <stdio.h>

int main() {
    int age = -1;

    printf("while版本：\n");
    while (age >= 0 && age <= 150) {
        printf("合法年龄: %d\n", age);
        break;
    }

    printf("do-while版本：\n");
    do {
        printf("请输入年龄(0-150): ");
        scanf("%d", &age);
    } while (age < 0 || age > 150);

    printf("合法年龄: %d\n", age);

    return 0;
}
循环类型 适用场景 特点
for 已知循环次数 结构清晰，变量作用域有限
while 不确定循环次数 先判断后执行，可能0次
do-while 至少执行一次 先执行后判断，至少1次
